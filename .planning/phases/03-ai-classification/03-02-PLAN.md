    ```python
    """
    transactions.py — Transactions page with AI classification CTA and inline review.

    Implements:
      - D-01: Manual classify button (user-initiated)
      - D-02: Idempotent — only classifies rows without category_id or confidence_score
                            and not manually_reviewed
      - D-03: Inline review via st.data_editor SelectboxColumn for low-confidence rows
      - D-04: No separate review page — all in this table
      - D-10/D-11: Only low-confidence rows show badge and correction column
    """
    import streamlit as st
    import pandas as pd
    from src.auth import get_current_user
    from src.config import get_authenticated_client
    from src.data.classifier import trigger_classification


    def _format_valor(amount: float) -> str:
        """Format float as Brazilian currency string."""
        return f"R$ {amount:,.2f}".replace(",", "X").replace(".", ",").replace("X", ".")


    def _confidence_label(score: str | None) -> str:
        """Return display label for confidence score."""
        if score == "high":
            return "Alta"
        elif score == "medium":
            return "Media"
        elif score == "low":
            return "Baixa"
        return "—"


    def show_transactions():
        """Render the Transactions page.

        Sections:
          1. Classify CTA button (D-01)
          2. Transactions table with inline correction for low-confidence rows (D-03)
        """
        st.title("Transacoes")

        user = get_current_user()
        if not user:
            st.stop()

        client = get_authenticated_client()

        # ------------------------------------------------------------------
        # Section 1: Classification CTA (D-01)
        # ------------------------------------------------------------------

        # Count unclassified transactions (for button disabled state)
        unclassified_resp = (
            client.table("transactions")
            .select("id", count="exact")
            .eq("user_id", user["id"])
            .or_("category_id.is.null,confidence_score.is.null")
            .eq("manually_reviewed", False)
            .execute()
        )
        unclassified_count = unclassified_resp.count or 0

        # Surface classification result from previous run (persists across st.rerun)
        if st.session_state.get("classify_success"):
            msg = st.session_state.pop("classify_success")
            st.success(msg)
        if st.session_state.get("classify_error"):
            msg = st.session_state.pop("classify_error")
            st.error(msg)

        st.markdown('<div class="classify-btn">', unsafe_allow_html=True)
        classify_clicked = st.button(
            "Classificar transacoes nao classificadas",
            key="btn_classify",
            disabled=(unclassified_count == 0),
            use_container_width=True,
        )
        st.markdown("</div>", unsafe_allow_html=True)

        if unclassified_count == 0:
            st.caption("Todas as transacoes ja foram classificadas")

        if classify_clicked:
            with st.spinner("Classificando... Isso pode levar alguns segundos."):
                result = trigger_classification(client, user["id"])

            if result["success"]:
                count = result.get("classified_count", 0)
                st.session_state["classify_success"] = (
                    f"{count} transacoes classificadas com sucesso!"
                )
            else:
                st.session_state["classify_error"] = result.get(
                    "error", "Erro ao conectar com o servico de classificacao. Tente novamente."
                )
            st.rerun()

        st.divider()

        # ------------------------------------------------------------------
        # Section 2: Transactions table with inline correction (D-03, D-11)
        # ------------------------------------------------------------------

        # Fetch all transactions with category info via join
        txn_resp = (
            client.table("transactions")
            .select("id, date, description, merchant_name, amount, confidence_score, manually_reviewed, categories(name, emoji)")
            .eq("user_id", user["id"])
            .order("date", desc=True)
            .execute()
        )
        transactions = txn_resp.data or []

        if not transactions:
            st.markdown('<div class="empty-state">', unsafe_allow_html=True)
            st.markdown("### Nenhuma transacao encontrada")
            st.markdown("Gere dados de teste nas Configuracoes para comecar.")
            st.markdown("</div>", unsafe_allow_html=True)
            return

        # Fetch categories for correction selectbox
        cat_resp = (
            client.table("categories")
            .select("id, name, emoji")
            .order("is_default", desc=True)
            .order("name")
            .execute()
        )
        categories = cat_resp.data or []
        category_options = [""] + [f"{c['emoji']} {c['name']}" for c in categories]
        category_display_to_id = {f"{c['emoji']} {c['name']}": c["id"] for c in categories}

        # Build display DataFrame
        rows = []
        for t in transactions:
            cat = t.get("categories") or {}
            cat_name = f"{cat.get('emoji', '')} {cat.get('name', '')}".strip() if cat else "—"
            is_low = t.get("confidence_score") == "low" and not t.get("manually_reviewed", False)
            rows.append({
                "_id": t["id"],
                "_is_low": is_low,
                "Data": t["date"],
                "Descricao": t["description"],
                "Merchant": t.get("merchant_name") or "—",
                "Categoria": cat_name,
                "Valor": t["amount"],
                "Confianca": "? Baixa" if is_low else _confidence_label(t.get("confidence_score")),
                "Corrigir": "" if is_low else None,
            })

        df = pd.DataFrame(rows)

        # Show inline-correctable table only if there are low-confidence rows
        low_count = df["_is_low"].sum()

        if low_count > 0:
            st.markdown(
                f'<p style="color: #EF4444; font-size: 14px;">⚠️ {low_count} transacao(oes) com confianca baixa — selecione a categoria correta na coluna "Corrigir".</p>',
                unsafe_allow_html=True,
            )

        # Track already-corrected IDs in this session to avoid duplicate PATCHes
        if "corrected_ids" not in st.session_state:
            st.session_state["corrected_ids"] = set()

        # Editable table — Corrigir column only for low-confidence rows
        # Non-low rows have None in Corrigir which hides the selectbox
        st.markdown('<div class="transactions-container">', unsafe_allow_html=True)

        display_df = df[["Data", "Descricao", "Merchant", "Categoria", "Valor", "Confianca", "Corrigir"]].copy()

        edited_df = st.data_editor(
            display_df,
            column_config={
                "Data": st.column_config.DateColumn("Data", format="DD/MM/YYYY"),
                "Valor": st.column_config.NumberColumn("Valor", format="R$ %.2f"),
                "Categoria": st.column_config.TextColumn("Categoria", disabled=True),
                "Confianca": st.column_config.TextColumn("Confianca", disabled=True),
                "Descricao": st.column_config.TextColumn("Descricao", disabled=True),
                "Merchant": st.column_config.TextColumn("Merchant", disabled=True),
                "Corrigir": st.column_config.SelectboxColumn(
                    "Corrigir",
                    options=category_options,
                    required=False,
                    help="Selecione a categoria correta",
                ),
            },
            disabled=["Data", "Descricao", "Merchant", "Categoria", "Valor", "Confianca"],
            hide_index=True,
            use_container_width=True,
            key="transactions_editor",
        )

        st.markdown("</div>", unsafe_allow_html=True)

        # Detect and apply corrections
        correction_made = False
        for idx, row in edited_df.iterrows():
            # Only process rows that were low-confidence and have a non-empty Corrigir
            original_is_low = df.at[idx, "_is_low"]
            txn_id = df.at[idx, "_id"]
            corrigir_val = row.get("Corrigir", "")

            if (
                original_is_low
                and corrigir_val
                and corrigir_val != ""
                and txn_id not in st.session_state["corrected_ids"]
            ):
                new_cat_id = category_display_to_id.get(corrigir_val)
                if new_cat_id:
                    try:
                        client.table("transactions").update({
                            "category_id": new_cat_id,
                            "confidence_score": "high",
                            "manually_reviewed": True,
                        }).eq("id", txn_id).execute()
                        st.session_state["corrected_ids"].add(txn_id)
                        correction_made = True
                        st.toast("Categoria corrigida.", icon="✅")
                    except Exception:
                        st.error("Erro ao salvar correcao. Tente novamente.")

        if correction_made:
            st.rerun()
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Streamlit UI → Supabase (correction write) | User-supplied selectbox value mapped to category_id before PATCH |
| Streamlit session_state | corrected_ids set persists per session; cleared on page reload |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | inline correction PATCH | mitigate | category_id is resolved from category_display_to_id dict (server-sourced from Supabase) — user cannot inject an arbitrary UUID. Input validation: only options from the Supabase-fetched category list are accepted. |
| T-02-02 | Tampering | correction loop duplicate PATCH | mitigate | st.session_state["corrected_ids"] set prevents re-sending same PATCH on each Streamlit rerun. After correction, confidence_score="high" so row won't appear as low-confidence on next load. |
| T-02-03 | Denial of Service | classify button + Make.com | mitigate | Button disabled when unclassified_count == 0. trigger_classification() has timeout=60s. No retry loop. |
| T-02-04 | Information Disclosure | transaction data in browser | accept | Transactions are user's own data (RLS enforced via get_authenticated_client()). No cross-user data visible. |
| T-02-05 | Spoofing | user identity in classify trigger | mitigate | user_id passed to trigger_classification() comes from get_current_user() which reads st.session_state["access_token"] — not user input. JWT validated by Supabase on client creation. |
</threat_model>
