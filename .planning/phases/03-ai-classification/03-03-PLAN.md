    ```python
    """
    categories.py — Category CRUD operations for Phase 3.

    All operations use the authenticated Supabase client (RLS-scoped).
    Default categories (is_default=True) are visible but not modifiable per RLS.

    Implements AICL-05 + D-05 + D-06.
    """
    from __future__ import annotations
    from typing import Any


    def get_all_categories(client: Any) -> list[dict[str, Any]]:
        """Fetch all categories visible to the current user (default + user's own).

        Ordered: defaults first (is_default DESC), then alphabetically by name.

        Args:
            client: Authenticated Supabase client

        Returns:
            List of category dicts with id, name, slug, emoji, color_hex, is_default, user_id
        """
        resp = (
            client.table("categories")
            .select("id, name, slug, emoji, color_hex, is_default, user_id")
            .order("is_default", desc=True)
            .order("name")
            .execute()
        )
        return resp.data or []


    def add_category(
        client: Any,
        user_id: str,
        name: str,
        color_hex: str,
        emoji: str,
    ) -> dict[str, Any]:
        """Create a new custom category for the given user.

        Performs case-insensitive duplicate name check before insert.
        Slug is derived from name (lowercase, spaces/hyphens to underscores).
        Emoji defaults to '🏷️' if empty string passed.

        Args:
            client: Authenticated Supabase client
            user_id: Supabase auth.uid() of the current user
            name: Display name for the category (max 50 chars)
            color_hex: Hex color string (e.g. "#FF5733") for Plotly charts
            emoji: Single emoji character, or empty string for default fallback

        Returns:
            {"success": True} on success
            {"success": False, "error": "message"} on duplicate name or DB error
        """
        name = name.strip()
        if not name:
            return {"success": False, "error": "O nome da categoria nao pode estar vazio."}

        # Case-insensitive duplicate check (D-06: prevent duplicate slugs)
        existing = (
            client.table("categories")
            .select("id")
            .ilike("name", name)
            .execute()
        )
        if existing.data:
            return {"success": False, "error": "Ja existe uma categoria com esse nome."}

        slug = name.lower().replace(" ", "_").replace("-", "_")
        resolved_emoji = emoji.strip() if emoji and emoji.strip() else "🏷️"

        try:
            client.table("categories").insert({
                "name": name,
                "slug": slug,
                "emoji": resolved_emoji,
                "color_hex": color_hex,
                "user_id": user_id,
                "is_default": False,
            }).execute()
            return {"success": True}
        except Exception as e:
            if "23505" in str(e):
                return {"success": False, "error": "Ja existe uma categoria com esse nome."}
            return {"success": False, "error": "Erro ao criar categoria."}


    def rename_category(
        client: Any,
        user_id: str,
        category_id: str,
        new_name: str,
    ) -> dict[str, Any]:
        """Rename a custom category.

        Only the owner can rename their own non-default categories (enforced by RLS).
        Performs case-insensitive duplicate name check before update.

        Args:
            client: Authenticated Supabase client
            user_id: Supabase auth.uid() of the current user (for duplicate check scope)
            category_id: UUID of the category to rename
            new_name: New display name (must not duplicate existing category name)

        Returns:
            {"success": True} on success
            {"success": False, "error": "message"} on duplicate or DB error
        """
        new_name = new_name.strip()
        if not new_name:
            return {"success": False, "error": "O nome da categoria nao pode estar vazio."}

        # Duplicate check: exclude current category from the check
        existing = (
            client.table("categories")
            .select("id")
            .ilike("name", new_name)
            .execute()
        )
        # Filter out the current category's own ID (renaming to same name is allowed)
        conflicts = [row for row in (existing.data or []) if row["id"] != category_id]
        if conflicts:
            return {"success": False, "error": "Ja existe uma categoria com esse nome."}

        try:
            client.table("categories").update({"name": new_name}).eq(
                "id", category_id
            ).eq("user_id", user_id).execute()
            return {"success": True}
        except Exception:
            return {"success": False, "error": "Erro ao renomear categoria."}


    def delete_category(
        client: Any,
        user_id: str,
        category_id: str,
    ) -> dict[str, Any]:
        """Delete a custom category.

        RLS policy categories_delete_own ensures only the owner can delete,
        and only non-default categories. Transactions linked to this category
        will have category_id set to NULL (ON DELETE SET NULL on FK).

        Args:
            client: Authenticated Supabase client
            user_id: Supabase auth.uid() of the current user
            category_id: UUID of the category to delete

        Returns:
            {"success": True} on success
            {"success": False, "error": "message"} on DB error
        """
        try:
            client.table("categories").delete().eq(
                "id", category_id
            ).eq("user_id", user_id).execute()
            return {"success": True}
        except Exception:
            return {"success": False, "error": "Erro ao remover categoria."}
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | add_category name input | mitigate | name.strip() applied; max_chars=50 on st.text_input; .ilike() duplicate check before insert; slug derived server-side, not user-supplied. |
| T-03-02 | Tampering | delete_category — targeting another user's category | mitigate | delete_category() calls .eq("user_id", user_id) as second filter. Supabase RLS categories_delete_own also enforces auth.uid() = user_id AND is_default = false. Double enforcement. |
| T-03-03 | Tampering | rename_category — targeting a default category | mitigate | RLS categories_update_own blocks UPDATE where is_default = true. Python Excluir button for defaults is disabled=True to prevent UI exposure. Defense in depth. |
| T-03-04 | Information Disclosure | color_hex injection via st.color_picker | accept | st.color_picker returns a validated hex string (#RRGGBB format). No arbitrary input. |
| T-03-05 | Tampering | emoji field — arbitrary unicode injection | mitigate | max_chars=4 on st.text_input prevents large strings. Emoji rendered via st.write() in display — no unsafe_allow_html. Category name rendered via st.write() not raw markdown. |
</threat_model>
